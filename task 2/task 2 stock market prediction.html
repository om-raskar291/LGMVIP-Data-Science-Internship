def dataset_created (dataset, time_step = 1):
    data_X, data_Y = [], []
    for i in range(int(len(dataset) - time_step-1)):
        a = dataset[i:(i + time_step), 0]
        data_X.append(a)
        data_Y.append(dataset[i + time_step, 0])
    return np.array(data_X), np.array(data_Y)
